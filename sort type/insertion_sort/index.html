const array = [10, 2, 4, 3, 7];


console.log(array); //[10, 2, 4, 3, 7];
const insertionSort = (arr) => {
const len = arr.length;
for (let i = 1; i < len; i++) {
let key = arr[i];
let j = i - 1;
while (j >= 0 && arr[j] > key) {
arr[j + 1] = arr[j];
j = j - 1;
}
arr[j + 1] = key;
}
return arr;
};
console.log(insertionSort(array)); // [2, 3, 4, 7, 10]
